1. Trabajando con Números

En JavaScript, podemos realizar operaciones matemáticas básicas como suma (+), resta (-), multiplicación (*) y división (/).

// Variables globales
let currentInput = '0';
let previousInput = '';
let operation = null;
let resetScreen = false;

// Actualizar la pantalla de la calculadora
function updateDisplay() {
    const display = document.getElementById('calc-display');
    display.textContent = currentInput;
}

// Agregar dígitos a la pantalla
function appendToDisplay(value) {
    if (currentInput === '0' || resetScreen) {
        currentInput = value;
        resetScreen = false;
    } else {
        currentInput += value;
    }
    updateDisplay();
}

// Establecer la operación
function setOperation(op) {
    if (currentInput === '') return;
    if (previousInput !== '') {
        calculate();
    }
    operation = op;
    previousInput = currentInput;
    resetScreen = true;
}

// Realizar el cálculo
function calculate() {
    let result;
    const prev = parseFloat(previousInput);
    const current = parseFloat(currentInput);
    
    if (isNaN(prev) || isNaN(current)) return;
    
    switch (operation) {
        case '+':
            result = prev + current;
            break;
        case '-':
            result = prev - current;
            break;
        case '*':
            result = prev * current;
            break;
        case '/':
            result = prev / current;
            break;
        case '%':
            result = prev % current;
            break;
        default:
            return;
    }
    
    currentInput = result.toString();
    operation = null;
    previousInput = '';
    resetScreen = true;
    updateDisplay();
}

// Limpiar la pantalla
function clearDisplay() {
    currentInput = '0';
    previousInput = '';
    operation = null;
    updateDisplay();
}

// Inicializar la calculadora
updateDisplay();

Conversión de Tipos

Es importante recordar que los valores de los inputs son cadenas de texto por defecto. Para realizar operaciones matemáticas, necesitamos convertirlos a números:

Convertir Texto a Número

Métodos para convertir texto a número:
  • parseInt() - Convierte una cadena en un número entero
  • parseFloat() - Convierte una cadena en un número decimal
  • Operador + - Convierte una cadena en número si es posible
  • Number() - Convierte un valor en un número
// Ejemplos:
let texto1 = "10";
let texto2 = "5.5";

// Usando parseInt() - convierte a entero
let entero = parseInt(texto1); // 10

// Usando parseFloat() - convierte a decimal
let decimal = parseFloat(texto2); // 5.5

// Usando el operador +
let num1 = +texto1; // 10
let num2 = +texto2; // 5.5

// Usando Number()
let numero1 = Number(texto1); // 10
let numero2 = Number(texto2); // 5.5

2. Validación de Números

Es importante validar que los valores ingresados sean números válidos antes de realizar operaciones con ellos.

Validación de Números

Por favor, ingresa una edad válida (entre 1 y 120)
Métodos útiles para validar números:
  • isNaN(valor) - Verifica si un valor NO es un número
  • Number.isInteger(valor) - Verifica si un valor es un número entero
  • typeof valor === 'number' - Verifica el tipo de dato
// Función para validar si un valor es un número válido
function esNumeroValido(valor) {
    // Verificar si el valor es un número y no es NaN
    return !isNaN(valor) && 
           valor !== null && 
           valor !== '' && 
           valor !== true && 
           valor !== false;
}

// Ejemplo de uso
let valor1 = "25";
let valor2 = "abc";

console.log(esNumeroValido(valor1)); // true
console.log(esNumeroValido(valor2)); // false

Manejo de Errores

Es importante manejar posibles errores al trabajar con operaciones matemáticas, especialmente la división por cero.

División con Manejo de Errores

function dividirNumeros() {
    // Obtener valores de los inputs
    const dividendo = parseFloat(document.getElementById('dividendo').value);
    const divisor = parseFloat(document.getElementById('divisor').value);
    const resultadoElemento = document.getElementById('resultado-division');
    
    // Validar que ambos campos tengan valores
    if (isNaN(dividendo) || isNaN(divisor)) {
        resultadoElemento.textContent = 'Por favor, ingresa números válidos en ambos campos';
        resultadoElemento.style.color = 'red';
        return;
    }
    
    // Verificar división por cero
    if (divisor === 0) {
        resultadoElemento.textContent = 'Error: No se puede dividir por cero';
        resultadoElemento.style.color = 'red';
        return;
    }
    
    // Realizar la división y mostrar el resultado
    const resultado = dividendo / divisor;
    resultadoElemento.textContent = `Resultado: ${resultado.toFixed(2)}`;
    resultadoElemento.style.color = 'green';
}
