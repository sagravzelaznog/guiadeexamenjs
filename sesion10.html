Cómo jugar

Instrucciones del juego

El objetivo del juego es adivinar un número secreto entre 1 y 100. Sigue estas pistas para encontrarlo:

  1. Ingresa un número en el campo de texto y presiona "Adivinar".
  2. Recibirás pistas si el número es mayor o menor al número secreto.
  3. ¡Intenta adivinarlo en la menor cantidad de intentos posible!
  4. El juego lleva registro de tus intentos y el tiempo que tardas.
Código de ejemplo
// Generar número aleatorio entre 1 y 100
const numeroSecreto = Math.floor(Math.random() * 100) + 1;

// Función para verificar el intento
function verificarIntento(intento) {
    if (intento === numeroSecreto) {
        return 'correcto';
    } else if (intento < numeroSecreto) {
        return 'mayor';
    } else {
        return 'menor';
    }
}

// Ejemplo de uso
let miIntento = 50;
let resultado = verificarIntento(miIntento);
console.log(`Tu número es ${resultado} que el número secreto.`);

Código del juego
// Variables del juego
let secretNumber;
let attempts;
let startTime;
let timerInterval;
let gameActive = false;

// Elementos del DOM
const guessInput = document.getElementById('guessInput');
const guessButton = document.getElementById('guessButton');
const feedback = document.getElementById('feedback');
const attemptsCount = document.getElementById('attemptsCount');
const guessList = document.getElementById('guessList');
const timerDisplay = document.getElementById('timerDisplay');

// Iniciar nuevo juego
function startNewGame() {
    // Generar número secreto
    secretNumber = Math.floor(Math.random() * 100) + 1;
    attempts = 0;
    gameActive = true;
    
    // Reiniciar interfaz
    guessInput.value = '';
    guessList.innerHTML = '';
    feedback.textContent = '';
    feedback.className = 'feedback';
    updateAttempts();
    
    // Iniciar temporizador
    startTimer();
    
    // Habilitar entrada
    guessInput.disabled = false;
    guessButton.disabled = false;
    
    console.log('Número secreto:', secretNumber); // Para depuración
}

// Verificar el intento del jugador
function checkGuess() {
    if (!gameActive) return;
    
    const guess = parseInt(guessInput.value);
    
    // Validar entrada
    if (isNaN(guess) || guess < 1 || guess > 100) {
        showFeedback('Por favor, ingresa un número entre 1 y 100', 'error');
        return;
    }
    
    // Incrementar intentos
    attempts++;
    updateAttempts();
    
    // Verificar el número
    if (guess === secretNumber) {
        gameWon();
    } else {
        const hint = guess < secretNumber ? 'mayor' : 'menor';
        showFeedback(`El número es ${hint} que ${guess}`, hint === 'mayor' ? 'info' : 'warning');
        addGuessToHistory(guess, hint);
    }
    
    // Limpiar el campo de entrada
    guessInput.value = '';
    guessInput.focus();
}

// Mostrar mensaje de retroalimentación
function showFeedback(message, type) {
    feedback.textContent = message;
    feedback.className = `feedback show ${type}`;
}

// Actualizar contador de intentos
function updateAttempts() {
    attemptsCount.textContent = attempts;
}

// Añadir intento al historial
function addGuessToHistory(guess, hint) {
    const li = document.createElement('div');
    li.className = 'history-item';
    
    const guessSpan = document.createElement('span');
    guessSpan.className = 'history-number';
    guessSpan.textContent = guess;
    
    const hintSpan = document.createElement('span');
    hintSpan.className = `history-hint ${hint}`;
    
    const icon = document.createElement('i');
    icon.className = `fas fa-arrow-${hint === 'mayor' ? 'up' : 'down'} hint-icon ${hint}`;
    
    hintSpan.appendChild(icon);
    hintSpan.appendChild(document.createTextNode(` El número es ${hint}`));
    
    li.appendChild(guessSpan);
    li.appendChild(hintSpan);
    
    guessList.prepend(li);
}

// Juego ganado
function gameWon() {
    gameActive = false;
    clearInterval(timerInterval);
    
    // Mostrar mensaje de victoria
    showFeedback(`¡Correcto! Has adivinado en ${attempts} intentos.`, 'success');
    
    // Mostrar animación de celebración
    celebrate();
    
    // Deshabilitar entrada
    guessInput.disabled = true;
    guessButton.disabled = true;
    
    // Mostrar modal de victoria
    showVictoryModal();
}

// Iniciar temporizador
function startTimer() {
    startTime = new Date();
    clearInterval(timerInterval);
    
    timerInterval = setInterval(() => {
        const now = new Date();
        const elapsed = Math.floor((now - startTime) / 1000);
        const minutes = Math.floor(elapsed / 60).toString().padStart(2, '0');
        const seconds = (elapsed % 60).toString().padStart(2, '0');
        timerDisplay.textContent = `${minutes}:${seconds}`;
    }, 1000);
}

// Animación de celebración
function celebrate() {
    const celebration = document.getElementById('celebration');
    const message = document.getElementById('celebrationMessage');
    
    // Mostrar mensaje
    setTimeout(() => {
        message.classList.add('show');
    }, 500);
    
    // Crear confeti
    for (let i = 0; i < 100; i++) {
        createConfetti();
    }
    
    celebration.classList.add('active');
    
    // Ocultar después de 5 segundos
    setTimeout(() => {
        celebration.classList.remove('active');
        message.classList.remove('show');
    }, 5000);
}

// Crear confeti
function createConfetti() {
    const confetti = document.createElement('div');
    confetti.className = 'confetti';
    
    // Posición aleatoria
    const posX = Math.random() * 100;
    const size = Math.random() * 10 + 5;
    const duration = Math.random() * 3 + 2;
    const delay = Math.random() * 5;
    
    // Color aleatorio
    const colors = ['#ff0000', '#00ff00', '#0000ff', '#ffff00', '#00ffff', '#ff00ff'];
    const color = colors[Math.floor(Math.random() * colors.length)];
    
    // Estilos
    confetti.style.left = `${posX}%`;
    confetti.style.width = `${size}px`;
    confetti.style.height = `${size}px`;
    confetti.style.background = color;
    confetti.style.animationDuration = `${duration}s`;
    confetti.style.animationDelay = `${delay}s`;
    
    document.getElementById('celebration').appendChild(confetti);
    
    // Eliminar después de la animación
    setTimeout(() => {
        confetti.remove();
    }, (duration + delay) * 1000);
}

// Mostrar modal de victoria
function showVictoryModal() {
    const modal = document.getElementById('victoryModal');
    const time = timerDisplay.textContent;
    const statsText = `Completado en ${attempts} intentos y ${time} minutos.`;
    
    document.getElementById('victoryStats').textContent = statsText;
    modal.classList.add('show');
    
    // Guardar mejor puntuación
    saveBestScore(attempts, time);
}

// Guardar mejor puntuación
function saveBestScore(attempts, time) {
    const bestScore = localStorage.getItem('bestScore');
    const currentScore = { attempts, time, date: new Date().toLocaleDateString() };
    
    if (!bestScore || JSON.parse(bestScore).attempts > attempts) {
        localStorage.setItem('bestScore', JSON.stringify(currentScore));
        updateBestScoreDisplay(currentScore);
    } else {
        updateBestScoreDisplay(JSON.parse(bestScore));
    }
}

// Actualizar visualización de la mejor puntuación
function updateBestScoreDisplay(score) {
    const bestScoreElement = document.getElementById('bestScore');
    bestScoreElement.textContent = `${score.attempts} (${score.time})`;
    bestScoreElement.title = `Logrado el ${score.date}`;
}

guessInput.addEventListener('keypress', (e) => {

// Dar una pista
function giveHint() {
    if (!gameActive) return;
    
    // Dar una pista cada 3 intentos
    if (attempts % 3 === 0) {
        const range = Math.floor(100 / (attempts / 3 + 1));
        const min = Math.max(1, secretNumber - range);
        const max = Math.min(100, secretNumber + range);
        
        showFeedback(`Pista: El número está entre ${min} y ${max}`, 'info');
    } else {
        showFeedback(`Sigue intentando. Pista disponible en ${3 - (attempts % 3)} intentos.`, 'warning');
    }
}

// Copiar código al portapapeles
function copyCode(elementId) {
    const codeElement = document.getElementById(elementId);
    const range = document.createRange();
    range.selectNode(codeElement);
    window.getSelection().removeAllRanges();
    window.getSelection().addRange(range);
    document.execCommand('copy');
    window.getSelection().removeAllRanges();
    
    // Mostrar notificación
    const notification = document.createElement('div');
    notification.textContent = '¡Código copiado!';
    notification.style.position = 'fixed';
    notification.style.bottom = '20px';
    notification.style.left = '50%';
    notification.style.transform = 'translateX(-50%)';
    notification.style.padding = '10px 20px';
    notification.style.background = '#28a745';
    notification.style.color = 'white';
    notification.style.borderRadius = '5px';
    notification.style.zIndex = '1000';
    notification.style.boxShadow = '0 2px 10px rgba(0,0,0,0.2)';
    notification.style.animation = 'fadeInOut 2s ease-in-out';
    
    document.body.appendChild(notification);
    
    setTimeout(() => {
        notification.remove();
    }, 2000);
}

// Iniciar el juego al cargar la página
window.addEventListener('load', () => {
    // Cargar mejor puntuación guardada
    const savedScore = localStorage.getItem('bestScore');
    if (savedScore) {
        updateBestScoreDisplay(JSON.parse(savedScore));
    }
    
    // Iniciar nuevo juego
    startNewGame();
    
    // Agregar estilos para la animación de notificación
    const style = document.createElement('style');
    style.textContent = `
        @keyframes fadeInOut {
            0% { opacity: 0; transform: translate(-50%, 20px); }
            15% { opacity: 1; transform: translate(-50%, 0); }
            85% { opacity: 1; transform: translate(-50%, 0); }
            100% { opacity: 0; transform: translate(-50%, -20px); }
        }
    `;
    document.head.appendChild(style);
});
